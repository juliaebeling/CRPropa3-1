207 Vector3d JF12Field::getDiskField(const double& r, const double& z, const double& phi, const double& sinPhi, const double& cosPhi) const {
208  Vector3d b(0.);
209  if (useDiskField) {
210  double lfDisk = logisticFunction(z, hDisk, wDisk);
211  if (r > 3 * kpc) {
212  double bMag;
213  if (r < 5 * kpc) {
214  // molecular ring
215  bMag = bRing * (5 * kpc / r) * (1 - lfDisk);
216  b.x += -bMag * sinPhi;
217  b.y += bMag * cosPhi;
218  } else {
219  // spiral region
220  double r_negx = r * exp(-(phi - M_PI) / tan90MinusPitch);
221  if (r_negx > rArms[7])
222  r_negx = r * exp(-(phi + M_PI) / tan90MinusPitch);
223  if (r_negx > rArms[7])
224  r_negx = r * exp(-(phi + 3 * M_PI) / tan90MinusPitch);
225 
226  for (int i = 7; i >= 0; i--)
227  if (r_negx < rArms[i])
228  bMag = bDisk[i];
229 
230  bMag *= (5 * kpc / r) * (1 - lfDisk);
231  b.x += bMag * (sinPitch * cosPhi - cosPitch * sinPhi);
232  b.y += bMag * (sinPitch * sinPhi + cosPitch * cosPhi);
233  }
234  }
235  }
236  return b;

239 Vector3d JF12Field::getToroidalHaloField(const double& r, const double& z, const double& sinPhi, const double& cosPhi) const {
240  Vector3d b(0.);

257 Vector3d JF12Field::getXField(const double& r, const double& z, const double& sinPhi, const double& cosPhi) const {
258  Vector3d b(0.);
259 
260  if (useXField && (r * r + z * z > 1 * kpc * kpc)){
261  double bMagX;
262  double sinThetaX, cosThetaX;
263  double rp;
264  double rc = rXc + fabs(z) / tanThetaX0;
265  if (r < rc) {
266  // varying elevation region
267  rp = r * rXc / rc;
268  bMagX = bX * exp(-1 * rp / rX) * pow(rXc / rc, 2.);
269  double thetaX = atan2(fabs(z), (r - rp));
270  if (z == 0)
271  thetaX = M_PI / 2.;
272  sinThetaX = sin(thetaX);
273  cosThetaX = cos(thetaX);
274  } else {
275  // constant elevation region
276  rp = r - fabs(z) / tanThetaX0;
277  bMagX = bX * exp(-rp / rX) * (rp / r);
278  sinThetaX = sinThetaX0;
279  cosThetaX = cosThetaX0;
280  }
281  double zsign = z < 0 ? -1 : 1;
282  b.x += zsign * bMagX * cosThetaX * cosPhi;
283  b.y += zsign * bMagX * cosThetaX * sinPhi;
284  b.z += bMagX * sinThetaX;
285  }
286  return b;
287 }

294 double JF12Field::getTurbulentStrength(const Vector3d& pos) const {
295  if (pos.getR() > 20 * kpc)
296  return 0;
297 
298  double r = sqrt(pos.x * pos.x + pos.y * pos.y); // in-plane radius
299  double phi = pos.getPhi(); // azimuth
300 
301  // disk
302  double bDisk = 0;
303  if (r < 5 * kpc) {
304  bDisk = bDiskTurb5;
305  } else {
306  // spiral region
307  double r_negx = r * exp(-(phi - M_PI) / tan90MinusPitch);
308  if (r_negx > rArms[7])
309  r_negx = r * exp(-(phi + M_PI) / tan90MinusPitch);
310  if (r_negx > rArms[7])
311  r_negx = r * exp(-(phi + 3 * M_PI) / tan90MinusPitch);
312 
313  for (int i = 7; i >= 0; i--)
314  if (r_negx < rArms[i])
315  bDisk = bDiskTurb[i];
316 
317  bDisk *= (5 * kpc) / r;
318  }
319  bDisk *= exp(-0.5 * pow(pos.z / zDiskTurb, 2));
320 
321  // halo
322  double bHalo = bHaloTurb * exp(-r / rHaloTurb)
323  * exp(-0.5 * pow(pos.z / zHaloTurb, 2));
324 
325  // modulate turbulent field
326  return sqrt(pow(bDisk, 2) + pow(bHalo, 2));
327 }
