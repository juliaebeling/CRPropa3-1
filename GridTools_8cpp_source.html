88  size_t Ny = grid->getNy();
89  size_t Nz = grid->getNz();
90  if ((Nx != Ny) or (Ny != Nz))
91  throw std::runtime_error("turbulentField: only cubic grid supported");
92 
93  Vector3d spacing = grid->getSpacing();
94  if ((spacing.x != spacing.y) or (spacing.y != spacing.z))
95  throw std::runtime_error("turbulentField: only equal spacing suported");
96 
97  if (lMin < 2 * spacing.x)
98  throw std::runtime_error("turbulentField: lMin < 2 * spacing");
99  if (lMin >= lMax)
100  throw std::runtime_error("turbulentField: lMin >= lMax");
101  if (lMax > Nx * spacing.x / 2)
102  throw std::runtime_error("turbulentField: lMax > size / 2");
103 
104  size_t n = Nx; // size of array
105  size_t n2 = (size_t) floor(n / 2) + 1; // size array in z-direction in configuration space
106 
107  // arrays to hold the complex vector components of the B(k)-field
108  fftwf_complex *Bkx, *Bky, *Bkz;
109  Bkx = (fftwf_complex*) fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
110  Bky = (fftwf_complex*) fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
111  Bkz = (fftwf_complex*) fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
112 
113 
114  Random random;
115  if (seed != 0)
116  random.seed(seed); // use given seed
117 
118  // calculate the n possible discrete wave numbers
119  double K[n];
120  for (int i = 0; i < n; i++)
121  K[i] = (double) i / n - i / (n / 2);
122 
123  // construct the field in configuration space
124  int i;
125  double k;
126 
127  // only used if there is a helicity
128  double Bktot, Bkplus, Bkminus, thetaplus, thetaminus;
129 
130  // parameters goes for non helical calculations
131  double theta, phase, cosPhase, sinPhase;
132 
133  double kMin = spacing.x / lMax;
134  double kMax = spacing.x / lMin;
135  Vector3f b; // real b-field vector
136  Vector3f ek, e1, e2; // orthogonal base
137  Vector3f n0(1, 1, 1); // arbitrary vector to construct orthogonal base
138 
139  for (size_t ix = 0; ix < n; ix++) {
140  for (size_t iy = 0; iy < n; iy++) {
141  for (size_t iz = 0; iz < n2; iz++) {
142 
143  i = ix * n * n2 + iy * n2 + iz;
144  ek.setXYZ(K[ix], K[iy], K[iz]);
145  k = ek.getR();
146 
147  // wave outside of turbulent range -> B(k) = 0
148  if ((k < kMin) || (k > kMax)) {
149  Bkx[i][0] = 0;
150  Bkx[i][1] = 0;
151  Bky[i][0] = 0;
152  Bky[i][1] = 0;
153  Bkz[i][0] = 0;
154  Bkz[i][1] = 0;
155  continue;
156  }
157 
158  // construct an orthogonal base ek, e1, e2
159  // (for helical fields together with the real transform the following convention
160  // must be used: e1(-k) = e1(k), e2(-k) = - e2(k)
161  if (helicity == true) {
162  if (ek.getAngleTo(n0) < 1e-3) { // ek parallel to (1,1,1)
163  e1.setXYZ(-1, 1, 0);
164  e2.setXYZ(1, 1, -2);
165  } else { // ek not parallel to (1,1,1)
166  e1 = n0.cross(ek);
167  e2 = ek.cross(e1);
168  }
169  e1 /= e1.getR();
170  e2 /= e2.getR();
171 
172 
173  double Bkprefactor = mu0 / (4 * M_PI * pow(k, 3));
174  Bktot = fabs(random.randNorm() * pow(k, alpha / 2));
175  Bkplus = Bkprefactor * sqrt((1 + H) / 2) * Bktot;
176  Bkminus = Bkprefactor * sqrt((1 - H) / 2) * Bktot;
177  thetaplus = 2 * M_PI * random.rand();
178  thetaminus = 2 * M_PI * random.rand();
179  double ctp = cos(thetaplus);
180  double stp = sin(thetaplus);
181  double ctm = cos(thetaminus);
182  double stm = sin(thetaminus);
183 
184  Bkx[i][0] = ((Bkplus * ctp + Bkminus * ctm) * e1.x + (-Bkplus * stp + Bkminus * stm) * e2.x) / sqrt(2);
185  Bkx[i][1] = ((Bkplus * stp + Bkminus * stm) * e1.x + ( Bkplus * ctp - Bkminus * ctm) * e2.x) / sqrt(2);
186  Bky[i][0] = ((Bkplus * ctp + Bkminus * ctm) * e1.y + (-Bkplus * stp + Bkminus * stm) * e2.y) / sqrt(2);
187  Bky[i][1] = ((Bkplus * stp + Bkminus * stm) * e1.y + ( Bkplus * ctp - Bkminus * ctm) * e2.y) / sqrt(2);
188  Bkz[i][0] = ((Bkplus * ctp + Bkminus * ctm) * e1.z + (-Bkplus * stp + Bkminus * stm) * e2.z) / sqrt(2);
189  Bkz[i][1] = ((Bkplus * stp + Bkminus * stm) * e1.z + ( Bkplus * ctp - Bkminus * ctm) * e2.z) / sqrt(2);
190 
191  Vector3f BkRe(Bkx[i][0], Bky[i][0], Bkz[i][0]);
192  Vector3f BkIm(Bkx[i][1], Bky[i][1], Bkz[i][1]);
193 
194  } else { // no helicity
195  if (ek.isParallelTo(n0, float(1e-3))) {
196  // ek parallel to (1,1,1)
197  e1.setXYZ(-1., 1., 0);
198  e2.setXYZ(1., 1., -2.);
199  } else {
200  // ek not parallel to (1,1,1)
201  e1 = n0.cross(ek);
202  e2 = ek.cross(e1);
203  }
204  e1 /= e1.getR();
205  e2 /= e2.getR();
206 
207  // random orientation perpendicular to k
208  theta = 2 * M_PI * random.rand();
209  b = e1 * cos(theta) + e2 * sin(theta);
210 
211  // normal distributed amplitude with mean = 0 and sigma = k^alpha/2
212  b *= random.randNorm() * pow(k, alpha / 2);
213 
214  // uniform random phase
215  phase = 2 * M_PI * random.rand();
216  cosPhase = cos(phase); // real part
217  sinPhase = sin(phase); // imaginary part
218 
219  Bkx[i][0] = b.x * cosPhase;
220  Bkx[i][1] = b.x * sinPhase;
221  Bky[i][0] = b.y * cosPhase;
222  Bky[i][1] = b.y * sinPhase;
223  Bkz[i][0] = b.z * cosPhase;
224  Bkz[i][1] = b.z * sinPhase;
225 
226  } // non helical case
227  } // for iz
228  } // for iy
229  } // for ix
230 
231 
232  // in-place, complex to real, inverse Fourier transformation on each component
233  // note that the last elements of B(x) are unused now
234  float *Bx = (float*) Bkx;
235  fftwf_plan plan_x = fftwf_plan_dft_c2r_3d(n, n, n, Bkx, Bx, FFTW_ESTIMATE);
236  fftwf_execute(plan_x);
237  fftwf_destroy_plan(plan_x);
238 
239  float *By = (float*) Bky;
240  fftwf_plan plan_y = fftwf_plan_dft_c2r_3d(n, n, n, Bky, By, FFTW_ESTIMATE);
241  fftwf_execute(plan_y);
242  fftwf_destroy_plan(plan_y);
243 
244  float *Bz = (float*) Bkz;
245  fftwf_plan plan_z = fftwf_plan_dft_c2r_3d(n, n, n, Bkz, Bz, FFTW_ESTIMATE);
246  fftwf_execute(plan_z);
247  fftwf_destroy_plan(plan_z);
248 
249  // save to grid
250  for (size_t ix = 0; ix < n; ix++) {
251  for (size_t iy = 0; iy < n; iy++) {
252  for (size_t iz = 0; iz < n; iz++) {
253  i = ix * n * 2 * n2 + iy * 2 * n2 + iz;
254  Vector3f &b = grid->get(ix, iy, iz);
255  b.x = Bx[i];
256  b.y = By[i];
257  b.z = Bz[i];
258  }
259  }
260  }

263  fftwf_free(Bky);
264  fftwf_free(Bkz);

267 }
