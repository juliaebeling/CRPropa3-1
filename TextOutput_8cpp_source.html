276 void TextOutput::load(const std::string &filename, ParticleCollector *collector){
277 
278  std::string line;
279  std::istream *in;
280  std::ifstream infile(filename.c_str());
281 
282  double lengthScale = Mpc; // default Mpc
283  double energyScale = EeV; // default EeV
284 
285  if (!infile.good())
286  throw std::runtime_error("crpropa::TextOutput: could not open file " + filename);
287  in = &infile;
288 
289  if (kiss::ends_with(filename, ".gz")){
290 #ifdef CRPROPA_HAVE_ZLIB
291  in = new zstream::igzstream(*in);
292 #else
293  throw std::runtime_error("CRPropa was build without Zlib compression!");
294 #endif
295  }
296 
297  while (std::getline(*in,line)) {
298  std::stringstream stream(line);
299  if (stream.peek() == '#')
300  continue;
301 
302  ref_ptr<Candidate> c = new Candidate();
303  double val_d; int val_i;
304  double x, y, z;
305  stream >> val_d;
306  c->setTrajectoryLength(val_d*lengthScale); // D
307  stream >> val_d;
308  c->setRedshift(val_d); // z
309  stream >> val_i;
310  c->setSerialNumber(val_i); // SN
311  stream >> val_i;
312  c->current.setId(val_i); // ID
313  stream >> val_d;
314  c->current.setEnergy(val_d*energyScale); // E
315  stream >> x >> y >> z;
316  c->current.setPosition(Vector3d(x, y, z)*lengthScale); // X, Y, Z
317  stream >> x >> y >> z;
318  c->current.setDirection(Vector3d(x, y, z)*lengthScale); // Px, Py, Pz
319  stream >> val_i; // SN0 (TODO: Reconstruct the parent-child relationship)
320  stream >> val_i;
321  c->source.setId(val_i); // ID0
322  stream >> val_d;
323  c->source.setEnergy(val_d*energyScale); // E0
324  stream >> x >> y >> z;
325  c->source.setPosition(Vector3d(x, y, z)*lengthScale); // X0, Y0, Z0
326  stream >> x >> y >> z;
327  c->source.setDirection(Vector3d(x, y, z)*lengthScale); // P0x, P0y, P0z
328  stream >> val_i; // SN1
329  stream >> val_i;
330  c->created.setId(val_i); // ID1
331  stream >> val_d;
332  c->created.setEnergy(val_d*energyScale); // E1
333  stream >> x >> y >> z;
334  c->created.setPosition(Vector3d(x, y, z)*lengthScale); // X1, Y1, Z1
335  stream >> x >> y >> z;
336  c->created.setDirection(Vector3d(x, y, z)*lengthScale); // P1x, P1y, P1z
337  stream >> val_d;
338  c->setWeight(val_d); // W
339 
340  collector->process(c);
341  }
342  infile.close();
343 }
