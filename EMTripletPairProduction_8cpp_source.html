99 void EMTripletPairProduction::performInteraction(Candidate *candidate) const {
100  // scale the particle energy instead of background photons
101  double z = candidate->getRedshift();
102  double E = candidate->current.getEnergy() * (1 + z);
103 
104  if (E < tabE.front() or E > tabE.back())
105  return;
106 
107  // sample the value of eps
108  Random &random = Random::instance();
109  size_t i = closestIndex(E, tabE);
110  size_t j = random.randBin(tabCDF[i]);
111  double s_kin = pow(10, log10(tabs[j]) + (random.rand() - 0.5) * 0.1);
112  double eps = s_kin / 4 / E; // random background photon energy
113 
114  // Use approximation from A. Mastichiadis et al., Astroph. Journ. 300:178-189 (1986), eq. 30.
115  // This approx is valid only for alpha >=100 where alpha = p0*eps*costheta - E0*eps
116  // For our purposes, me << E0 --> p0~E0 --> alpha = E0*eps*(costheta - 1) >= 100
117  double Epp = 5.7e-1 * pow(eps/mec2, -0.56) * pow(E/mec2, 0.44) * mec2;
118 
119  if (haveElectrons) {
120  Vector3d pos = random.randomInterpolatedPosition(candidate->previous.getPosition(), candidate->current.getPosition());
121  candidate->addSecondary( 11, Epp, pos);
122  candidate->addSecondary(-11, Epp, pos);
123  }
124 
125  // update the primary particle energy; do this after adding the secondaries to correctly set the secondaries parent
126  candidate->current.setEnergy((E - 2 * Epp));
127 }

129 void EMTripletPairProduction::process(Candidate *candidate) const {
130  // check if electron / positron
131  int id = candidate->current.getId();
132  if (abs(id) != 11)
133  return;
134 
135  // scale the particle energy instead of background photons
136  double z = candidate->getRedshift();
137  double E = (1 + z) * candidate->current.getEnergy();
138 
139  // check if in tabulated energy range
140  if (E < tabEnergy.front() or (E > tabEnergy.back()))
141  return;
142 
143  // cosmological scaling of interaction distance (comoving)
144  double scaling = pow(1 + z, 2) * photonFieldScaling(photonField, z);
145  double rate = scaling * interpolate(E, tabEnergy, tabRate);
146 
147  // check for interaction
148  Random &random = Random::instance();
149  double randDistance = -log(random.rand()) / rate;
150  if (candidate->getCurrentStep() > randDistance)
151  performInteraction(candidate);
152  else
153  candidate->limitNextStep(limit / rate);
154 }
