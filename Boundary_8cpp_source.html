51 void ReflectiveBox::process(Candidate *c) const {
52  Vector3d cur = (c->current.getPosition() - origin) / size; // current position in cell units
53  Vector3d n = cur.floor();
54 
55  if ((n.x == 0) and (n.y == 0) and (n.z == 0))
56  return; // do nothing if candidate is inside the box
57 
58  // flip direction
59  Vector3d nReflect(pow(-1, n.x), pow(-1, n.y), pow(-1, n.z));
60  c->current.setDirection(c->current.getDirection() * nReflect);
61  c->previous.setDirection(c->previous.getDirection() * nReflect);
62  c->created.setDirection(c->created.getDirection() * nReflect);
63  c->source.setDirection(c->source.getDirection() * nReflect);
64 
65  Vector3d src = (c->source.getPosition() - origin) / size; // initial position in cell units
66  Vector3d cre = (c->created.getPosition() - origin) / size; // initial position in cell units
67  Vector3d prv = (c->previous.getPosition() - origin) / size; // previous position in cell units
68 
69  // repeatedly translate until the current position is inside the cell
70  while ((cur.x < 0) or (cur.x > 1)) {
71  double t = 2 * (cur.x > 1);
72  src.x = t - src.x;
73  cre.x = t - cre.x;
74  prv.x = t - prv.x;
75  cur.x = t - cur.x;
76  }
77  while ((cur.y < 0) or (cur.y > 1)) {
78  double t = 2 * (cur.y > 1);
79  src.y = t - src.y;
80  cre.y = t - cre.y;
81  prv.y = t - prv.y;
82  cur.y = t - cur.y;
83  }
84  while ((cur.z < 0) or (cur.z > 1)) {
85  double t = 2 * (cur.z > 1);
86  src.z = t - src.z;
87  cre.z = t - cre.z;
88  prv.z = t - prv.z;
89  cur.z = t - cur.z;
90  }
91 
92  c->current.setPosition(cur * size + origin);
93  c->source.setPosition(src * size + origin);
94  c->created.setPosition(cre * size + origin);
95  c->previous.setPosition(prv * size + origin);
96 }

252 void CylindricalBoundary::process(Candidate *c) const {
253  Vector3d d = c->current.getPosition() - origin;
254  double R2 = pow(d.x, 2.)+pow(d.y, 2.);
255  double Z = fabs(d.z);
256  if ( R2 < pow(radius, 2.) and Z < height/2.) {
257  if(limitStep) {
258  c->limitNextStep(std::min(radius - pow(R2, 0.5), height/2. - Z) + margin);
259  }
