191  T interpolate(const Vector3d &position) const {
192  // position on a unit grid
193  Vector3d r = (position - gridOrigin) / spacing;
194 
195  // indices of lower and upper neighbors
196  int ix, iX, iy, iY, iz, iZ;
197  if (reflective) {
198  reflectiveClamp(r.x, Nx, ix, iX);
199  reflectiveClamp(r.y, Ny, iy, iY);
200  reflectiveClamp(r.z, Nz, iz, iZ);
201  } else {
202  periodicClamp(r.x, Nx, ix, iX);
203  periodicClamp(r.y, Ny, iy, iY);
204  periodicClamp(r.z, Nz, iz, iZ);
205  }
206 
207  // linear fraction to lower and upper neighbors
208  double fx = r.x - floor(r.x);
209  double fX = 1 - fx;
210  double fy = r.y - floor(r.y);
211  double fY = 1 - fy;
212  double fz = r.z - floor(r.z);
213  double fZ = 1 - fz;
214 
215  // trilinear interpolation (see http://paulbourke.net/miscellaneous/interpolation)
216  T b(0.);
217  //V000 (1 - x) (1 - y) (1 - z) +
218  b += get(ix, iy, iz) * fX * fY * fZ;
219  //V100 x (1 - y) (1 - z) +
220  b += get(iX, iy, iz) * fx * fY * fZ;
221  //V010 (1 - x) y (1 - z) +
222  b += get(ix, iY, iz) * fX * fy * fZ;
223  //V001 (1 - x) (1 - y) z +
224  b += get(ix, iy, iZ) * fX * fY * fz;
225  //V101 x (1 - y) z +
226  b += get(iX, iy, iZ) * fx * fY * fz;
227  //V011 (1 - x) y z +
228  b += get(ix, iY, iZ) * fX * fy * fz;
229  //V110 x y (1 - z) +
230  b += get(iX, iY, iz) * fx * fy * fZ;
231  //V111 x y z
232  b += get(iX, iY, iZ) * fx * fy * fz;
233
