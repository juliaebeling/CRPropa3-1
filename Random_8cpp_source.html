171 double Random::randPowerLaw(double index, double min, double max) {
172  if ((min < 0) || (max < min)) {
173  throw std::runtime_error(
174  "Power law distribution only possible for 0 <= min <= max");
175  }
176  //check for index -1!
177  if ((std::abs(index + 1.0)) < std::numeric_limits<double>::epsilon()) {
178  double part1 = log(max);
179  double part2 = log(min);
180  return exp((part1 - part2) * rand() + part2);
181  } else {
182  double part1 = pow(max, index + 1);
183  double part2 = pow(min, index + 1);
184  double ex = 1 / (index + 1);
185  return pow((part1 - part2) * rand() + part2, ex);
186  }
187 }

189 double Random::randBrokenPowerLaw(double index1, double index2,
190  double breakpoint, double min, double max) {
191  if ((min <= 0) || (max < min)) {
192  throw std::runtime_error(
193  "Power law distribution only possible for 0 < min <= max");
194  }
195  if (min >= breakpoint) {
196  return this->randPowerLaw(index2, min, max);
197  } else if (max <= breakpoint) {
198  return this->randPowerLaw(index2, min, max);
199  } else {
200  double intPL1;
201  // check if index1 = -1
202  if ((std::abs(index1 + 1.0)) < std::numeric_limits<double>::epsilon()) {
203  intPL1 = log(breakpoint / min);
204  } else {
205  intPL1 = (pow(breakpoint, index1 + 1) - pow(min, index1 + 1))
206  / (index1 + 1);
207  }
208  double intPL2;
209  // check if index2 = -1
210  if ((std::abs(index2 + 1.0)) < std::numeric_limits<double>::epsilon()) {
211  intPL2 = log(max / breakpoint) * pow(breakpoint, index1 - index2);
212  } else {
213  intPL2 = (pow(max, index2 + 1) - pow(breakpoint, index2 + 1))
214  * pow(breakpoint, index1 - index2) / (index2 + 1);
215  }
216  if (rand() > intPL1 / (intPL1 + intPL2))
217  return randPowerLaw(index2, breakpoint, max);
218  else
219  return randPowerLaw(index1, min, breakpoint);
220  }
221 }

244 uint32_t Random::randInt(const uint32_t& n) {
245 // Find which bits are used in n
246 // Optimized by Magnus Jonsson (magnus@smartelectronix.com)
247  uint32_t used = n;
248  used |= used >> 1;
249  used |= used >> 2;
250  used |= used >> 4;
251  used |= used >> 8;
252  used |= used >> 16;
253 
254 // Draw numbers until one is found in [0,n]
255  uint32_t i;
256  do
257  i = randInt() & used; // toss unused bits to shorten search
258  while (i > n);
259  return i;
260 }

298 void Random::seed(uint32_t * const bigSeed, const uint32_t seedLength) {
299 
300  initial_seed.resize(seedLength);
301  for (size_t i =0; i< seedLength; i++)
302  {
303  initial_seed[i] = bigSeed[i];
304  }
305 
306  initialize(19650218UL);
307  int i = 1;
308  uint32_t j = 0;
309  int k = (N > seedLength ? N : seedLength);
310  for (; k; --k) {
311  state[i] = state[i]
312  ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1664525UL);
313  state[i] += (bigSeed[j] & 0xffffffffUL) + j;
314  state[i] &= 0xffffffffUL;
315  ++i;
316  ++j;
317  if (i >= N) {
318  state[0] = state[N - 1];
319  i = 1;
320  }
321  if (j >= seedLength)
322  j = 0;
323  }
324  for (k = N - 1; k; --k) {
325  state[i] = state[i]
326  ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1566083941UL);
327  state[i] -= i;
328  state[i] &= 0xffffffffUL;
329  ++i;
330  if (i >= N) {
331  state[0] = state[N - 1];
332  i = 1;
333  }
334  }
335  state[0] = 0x80000000UL; // MSB is 1, assuring non-zero initial array
336  reload();
337 }

339 void Random::seed() {
340 // First try getting an array from /dev/urandom
341  FILE* urandom = std::fopen("/dev/urandom", "rb");
342  if (urandom) {
343  uint32_t bigSeed[N];
344  uint32_t *s = bigSeed;
345  int i = N;
346  bool success = true;
347  while (success && i--)
348  success = std::fread(s++, sizeof(uint32_t), 1, urandom) != 0;
349  std::fclose(urandom);
350  if (success) {
351  seed(bigSeed, N);
352  return;
353  }
354  }
355 
356 // Was not successful, so use time() and clock() instead
357  seed(hash(time(NULL), clock()));
358 }

384 uint32_t Random::hash(time_t t, clock_t c) {
385  static uint32_t differ = 0; // guarantee time-based seeds will change
386 
387  uint32_t h1 = 0;
388  unsigned char *p = (unsigned char *) &t;
389  for (size_t i = 0; i < sizeof(t); ++i) {
390  h1 *= std::numeric_limits<unsigned char>::max() + 2U;
391  h1 += p[i];
392  }
393  uint32_t h2 = 0;
394  p = (unsigned char *) &c;
395  for (size_t j = 0; j < sizeof(c); ++j) {
396  h2 *= std::numeric_limits<unsigned char>::max() + 2U;
397  h2 += p[j];
398  }
399  return (h1 + differ++) ^ h2;
400 }
