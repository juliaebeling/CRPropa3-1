30 void serialize(const string &filename, const ModelMatrixType& matrix)
31 {
32  ofstream outfile(filename.c_str(), ios::binary);
33  if (!outfile)
34  {
35  throw runtime_error("Can't write file: " + filename);
36  }
37 
38  uint32_t C = 0;
39  double val;
40 
41  C = (uint32_t) (matrix.nonZeros());
42  outfile.write((char*) &C, sizeof(uint32_t));
43  C = (uint32_t) (matrix.rows());
44  outfile.write((char*) &C, sizeof(uint32_t));
45  C = (uint32_t) (matrix.cols());
46  outfile.write((char*) &C, sizeof(uint32_t));
47 
48  // serialize non zero elements
49  for (size_t col_idx = 0; col_idx < matrix.cols(); ++col_idx)
50  {
51  for (ModelMatrixType::InnerIterator it(matrix,col_idx); it; ++it)
52  {
53  it.value();
54  C = (uint32_t) it.row();
55  outfile.write((char*) &C, sizeof(uint32_t));
56 
57  C = (uint32_t) it.col();
58  outfile.write((char*) &C, sizeof(uint32_t));
59 
60  val = it.value();
61  outfile.write((char*) &val, sizeof(double));
62  if (outfile.fail())
63  {
64  throw runtime_error("Error writing file: " + filename);
65  }
66  }
67  }
68 
69  outfile.close();
70 }

73 void deserialize(const string &filename, ModelMatrixType& matrix)
74 {
75  ifstream infile(filename.c_str(), ios::binary);
76  if (!infile)
77  {
78  throw runtime_error("Can't read file: " + filename);
79  }
80 
81  uint32_t nnz, nRows, nColumns;
82  infile.read((char*) &nnz, sizeof(uint32_t));
83  infile.read((char*) &nRows, sizeof(uint32_t));
84  infile.read((char*) &nColumns, sizeof(uint32_t));
85  matrix.resize(nRows, nColumns);
86  matrix.reserve(nnz);
87 
88  uint32_t row, column;
89  double val;
90  std::vector< Eigen::Triplet<double> > triplets;
91  triplets.resize(nnz);
92  for (size_t i = 0; i < nnz; i++)
93  {
94  infile.read((char*) &row, sizeof(uint32_t));
95  infile.read((char*) &column, sizeof(uint32_t));
96  infile.read((char*) &val, sizeof(double));
97  //M(size1,size2) = val;
98  triplets[i] = Eigen::Triplet<double>(row, column, val);
99  }
100  matrix.setFromTriplets(triplets.begin(), triplets.end());
101  matrix.makeCompressed();
102 }

138  void prod_up(const ModelMatrixType& matrix, double* model)
139 {
140 
141  // copy storage of model, as matrix vector product cannot be done
142  // in place
143 
144  const size_t mSize = matrix.cols();
145  double *origVectorStorage = new double[mSize];
146  memcpy(origVectorStorage, model, mSize * sizeof(double));

148 
149 
150  Eigen::Map<Eigen::Matrix<double, Eigen::Dynamic, 1> > origVectorAdaptor(origVectorStorage, mSize);
151  Eigen::Map<Eigen::Matrix<double, Eigen::Dynamic, 1> > modelVectorAdaptor(model, mSize);
152 
153  // perform the optimized product
154  modelVectorAdaptor = matrix * origVectorAdaptor;

156  // clean up
