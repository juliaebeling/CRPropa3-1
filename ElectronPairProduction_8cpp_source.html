80 double ElectronPairProduction::lossLength(int id, double lf, double z) const {
81  double Z = chargeNumber(id);
82  if (Z == 0)
83  return std::numeric_limits<double>::max(); // no pair production on uncharged particles
84 
85  lf *= (1 + z);
86  if (lf < tabLorentzFactor.front())
87  return std::numeric_limits<double>::max(); // below energy threshold
88 
89  double rate;
90  if (lf < tabLorentzFactor.back())
91  rate = interpolate(lf, tabLorentzFactor, tabLossRate); // interpolation
92  else
93  rate = tabLossRate.back() * pow(lf / tabLorentzFactor.back(), -0.6); // extrapolation
94 
95  double A = nuclearMass(id) / mass_proton; // more accurate than massNumber(Id)
96  rate *= Z * Z / A * pow(1 + z, 3) * photonFieldScaling(photonField, z);
97  return 1. / rate;
98 }

100 void ElectronPairProduction::process(Candidate *c) const {
101  int id = c->current.getId();
102  if (not (isNucleus(id)))
103  return; // only nuclei
104 
105  double lf = c->current.getLorentzFactor();
106  double z = c->getRedshift();
107  double losslen = lossLength(id, lf, z); // energy loss length
108  if (losslen >= std::numeric_limits<double>::max())
109  return;
110 
111  double step = c->getCurrentStep() / (1 + z); // step size in local frame
112  double loss = step / losslen; // relative energy loss
113 
114  if (haveElectrons) {
115  double dE = c->current.getEnergy() * loss; // energy loss
116  int i = round((log10(lf) - 6.05) * 10); // find closest cdf(Ee|log10(gamma))
117  i = std::min(std::max(i, 0), 69);
118  Random &random = Random::instance();
119 
120  // draw pairs as long as their energy is smaller than the pair production energy loss
121  while (dE > 0) {
122  size_t j = random.randBin(tabSpectrum[i]);
123  double Ee = pow(10, 6.95 + (j + random.rand()) * 0.1) * eV;
124  double Epair = 2 * Ee; // NOTE: electron and positron in general don't have same lab frame energy, but averaged over many draws the result is consistent
125  // if the remaining energy is not sufficient check for random accepting
126  if (Epair > dE)
127  if (random.rand() > (dE / Epair))
128  break; // not accepted
129 
130  // create pair and repeat with remaining energy
131  dE -= Epair;
132  Vector3d pos = random.randomInterpolatedPosition(c->previous.getPosition(), c->current.getPosition());
133  c->addSecondary( 11, Ee, pos);
134  c->addSecondary(-11, Ee, pos);
135  }
136  }
137 
138  c->current.setLorentzFactor(lf * (1 - loss));
139  c->limitNextStep(limit * losslen);
140 }
